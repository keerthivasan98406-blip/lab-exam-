1. Text Preprocessing Techniques

import nltk
from nltk.tokenize import word_tokenize
from nltk.corpus import stopwords
from nltk.stem import PorterStemmer, WordNetLemmatizer

nltk.download('punkt_tab')
nltk.download('stopwords')
nltk.download('wordnet')

text = "NLP helps computers understand language."

# Tokenization
tokens = word_tokenize(text)

# Stop-word removal
stop_words = set(stopwords.words('english'))
filtered = [word for word in tokens if word.lower() not in stop_words]

# Stemming
stemmer = PorterStemmer()
stemmed = [stemmer.stem(word) for word in filtered]

# Lemmatization
lemmatizer = WordNetLemmatizer()
lemmatized = [lemmatizer.lemmatize(word) for word in filtered]

print("Tokens:", tokens)
print("Filtered:", filtered)
print("Stemmed:", stemmed)
print("Lemmatized:", lemmatized)

import nltk

4. Named Entity Recognition (NER)

import nltk
from nltk import word_tokenize, pos_tag, ne_chunk
from nltk.tree import Tree

nltk.download('punkt')
nltk.download('averaged_perceptron_tagger')
nltk.download('maxent_ne_chunker_tab')
nltk.download('words')

text = "Google was founded by Larry Page in California."
tokens = word_tokenize(text)
pos_tags = pos_tag(tokens)
ner_tree = ne_chunk(pos_tags)

print("Named Entities:")
for subtree in ner_tree:
    if isinstance(subtree, Tree):
        entity = " ".join([word for word, tag in subtree.leaves()])
        label = subtree.label()
        print(entity, "->", label)

5. Text Classification using Machine Learning

from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB

texts = ["Win money now", "Limited offer", "Call me today", "Lets meet tomorrow"]
labels = ["spam", "spam", "ham", "ham"]

vectorizer = CountVectorizer()
X = vectorizer.fit_transform(texts)

model = MultinomialNB()
model.fit(X, labels)

sample = ["Free money offer"]
sample_vector = vectorizer.transform(sample)
prediction = model.predict(sample_vector)

print("Message:", sample[0])
print("Prediction:", prediction[0])

6. Sentiment Analysis

from textblob import TextBlob

reviews = ["I love this product", "This product is bad", "The product is okay"]

for review in reviews:
    blob = TextBlob(review)
    polarity = blob.sentiment.polarity
    if polarity > 0:
        sentiment = "Positive"
    elif polarity < 0:
        sentiment = "Negative"
    else:
        sentiment = "Neutral"
    print("Review:", review)
    print("Sentiment:", sentiment)
    print("Polarity:", polarity)
    print()

9.(i) Breadth First Search (BFS)

from collections import deque

graph = {
    'A': ['B', 'C'],
    'B': ['D'],
    'C': ['E'],
    'D': [],
    'E': []
}

def bfs(start):
    visited = []
    queue = deque([start])
    while queue:
        node = queue.popleft()
        if node not in visited:
            print(node, end=" ")
            visited.append(node)
            queue.extend(graph[node])

print("BFS Traversal:")
bfs('A')

9.(ii) DFS for Problem Solving

graph = {
    'A': ['B', 'C'],
    'B': ['D'],
    'C': ['E'],
    'D': [],
    'E': []
}

visited = []

def dfs(node):
    if node not in visited:
        print(node, end=" ")
        visited.append(node)
        for neighbour in graph[node]:
            dfs(neighbour)

print("DFS Traversal:")
dfs('A')

10.(i) Best-First Search (Heuristic-Based Search)

import heapq

heuristic = {'A': 5, 'B': 4, 'C': 2, 'D': 3, 'E': 1, 'F': 0}
graph = {'A': ['B', 'C'], 'B': ['D', 'E'], 'C': ['F'], 'D': [], 'E': ['F'], 'F': []}

def best_first_search(start, goal):
    visited = set()
    queue = []
    path = []
    heapq.heappush(queue, (heuristic[start], start))
    while queue:
        h_val, node = heapq.heappop(queue)
        if node not in visited:
            path.append(node)
            visited.add(node)
            if node == goal:
                break
            for neighbor in graph[node]:
                if neighbor not in visited:
                    heapq.heappush(queue, (heuristic[neighbor], neighbor))
    return path

print("Best-First Search Path:")
print(best_first_search('A', 'F'))
