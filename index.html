3. Implement form validation using JavaScript for user registration (name, email, password)

<!-- Implement form validation using JavaScript
     for user registration (name, email, password) -->
<!DOCTYPE html>
<html>
<head>
    <title>Form Validation</title>
</head>
<body>
    <h2>Registration Form</h2>
    <form onsubmit="return validate()">
        Name: <input type="text" id="name"><br><br>
        Email: <input type="text" id="email"><br><br>
        Password: <input type="password" id="password"><br><br>
        <input type="submit">
    </form>
    <script>
        function validate() {
            var name = document.getElementById("name").value.trim();
            var email = document.getElementById("email").value.trim();
            var password = document.getElementById("password").value;
            if (name == "") { alert("Enter Name"); return false; }
            if (email == "") { alert("Enter Email"); return false; }
            if (password.length < 6) {
                alert("Password must contain 6 characters");
                return false;
            }
            alert("Registration Successful");
            return true;
        }
    </script>
</body>
</html>

4. Develop a dynamic web page using JavaScript to perform DOM manipulation (add, edit, delete elements)

<!-- Develop a dynamic web page using JavaScript
     to perform DOM manipulation (add, edit, delete elements) -->
<!DOCTYPE html>
<html>
<body>
    <h3>Student List</h3>
    <input id="item" placeholder="Student name">
    <button id="add">Add</button>
    <button id="clear">Clear Text</button>
    <ul id="list"></ul>

    <script>
        const item = document.getElementById("item"),
              list = document.getElementById("list");

        function addItem() {
            const v = item.value.trim();
            if (!v) return;
            const li = document.createElement("li");
            const text = document.createElement("span");
            const edit = document.createElement("button");
            const del = document.createElement("button");
            text.textContent = v + " ";
            edit.textContent = "Edit";
            del.textContent = "Delete";
            edit.onclick = () => {
                const n = prompt("Edit name:", text.textContent.trim());
                if (n && n.trim()) text.textContent = n.trim() + " ";
            };
            del.onclick = () => li.remove();
            li.append(text, edit, document.createTextNode(" "), del);
            list.appendChild(li);
            item.value = "";
            item.focus();
        }

        document.getElementById("add").onclick = addItem;
        document.getElementById("clear").onclick = () => {
            item.value = ""; item.focus();
        };
        item.addEventListener("keydown", (e) => {
            if (e.key === "Enter") { e.preventDefault(); addItem(); }
        });
    </script>
</body>
</html>

    <div class="calculator">
        <h2>Calculator</h2>
        <input type="text" id="result" readonly>
        <div class="buttons">
            <button onclick="clearBox()">C</button>
            <button onclick="deleteLast()">DEL</button>
            <button onclick="display('%')" class="operator">%</button>
            <button onclick="display('/')" class="operator">/</button>
            <button onclick="display('7')">7</button>
            <button onclick="display('8')">8</button>
            <button onclick="display('9')">9</button>
            <button onclick="display('*')" class="operator">*</button>
            <button onclick="display('4')">4</button>
            <button onclick="display('5')">5</button>
            <button onclick="display('6')">6</button>
            <button onclick="display('-')" class="operator">-</button>
            <button onclick="display('1')">1</button>
            <button onclick="display('2')">2</button>
            <button onclick="display('3')">3</button>
            <button onclick="display('+')" class="operator">+</button>
            <button onclick="display('0')">0</button>
            <button onclick="display('.')">.</button>
            <button onclick="calculate()" class="equal">=</button>
        </div>
    </div>
    <script>
        function display(value) {
            document.getElementById("result").value += value;
        }
        function calculate() {
            var expression = document.getElementById("result").value;
            if (expression === "") return;
            try {
                document.getElementById("result").value = eval(expression);
            } catch (error) {
                document.getElementById("result").value = "Error";
            }
        }
        function clearBox() { document.getElementById("result").value = ""; }
        function deleteLast() {
            var current = document.getElementById("result").value;
            document.getElementById("result").value = current.slice(0, -1);
        }
    </script>
</body>
</html>

10. Develop a simple full-stack CRUD application to perform Create, Read, Update, and Delete operations

<!-- Develop a simple full-stack CRUD application
     to perform Create, Read, Update, and Delete operations -->
<!DOCTYPE html>
<html>
<head>
    <title>CRUD Application</title>
</head>
<body>
    <h2>CRUD Application</h2>
    <input type="text" id="name" placeholder="Enter Name">
    <button onclick="addData()">Add</button>
    <ul id="list"></ul>
    <script>
        var data = [];
        function showData() {
            document.getElementById("list").innerHTML = "";
            for (var i = 0; i < data.length; i++) {
                document.getElementById("list").innerHTML +=
                    "<li>" + data[i] +
                    " <button onclick='editData(" + i + ")'>Edit</button> " +
                    " <button onclick='deleteData(" + i + ")'>Delete</button></li>";
            }
        }
        function addData() {
            var name = document.getElementById("name").value;
            data.push(name);
            showData();
        }
        function editData(i) {
            var value = prompt("Edit Name", data[i]);
            data[i] = value;
            showData();
        }
        function deleteData(i) {
            data.splice(i, 1);
            showData();
        }
    </script>
</body>
</html>
